def generate_svr(x1, y1, kernel, extend=1.0, **kwargs):
    X = x1.reshape(-1,1)
    y = y1.reshape(-1,1)

    svr = SVR(kernel=kernel, **kwargs)
    model = svr.fit(X, y)

    x2 = np.linspace(X.min(), X.max()*extend, 100)
    y2 = model.predict(x2.reshape(-1,1))
    return x2, y2.flatten()

def plot_svr(x1, y1, kernel, **kwargs):
    x2, y2 = generate_svr(x1, y1, kernel, **kwargs)
    _, ax = plt.subplots(figsize=(12,7))
    ax.scatter(x1, y1, color='purple', marker='x', alpha=0.5)
    ax.plot(x2, y2, color='red', ls='-', lw=2)
    ax.grid(True)

In all of the cases tested, the rbf (radial basis function) has the best extrapolation properties. However, it does not perform as well as the exponential sine squared kernel for Gaussian Process Regression.

1.3 Effect of kernel choice when modeling functions that are predominantly linear¶
