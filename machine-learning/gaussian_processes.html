def generate_gpr(x1, y1, kernel, extend=1.0):
    X = x1.reshape(-1,1)
    y = y1.reshape(-1,1)

    GPR = GaussianProcessRegressor(kernel=kernel)
    model = GPR.fit(X, y)

    x2 = np.linspace(X.min(), X.max()*extend, 100)
    y2, y2_err = GPR.predict(x2.reshape(-1,1), return_std=True)
    return x2, y2.flatten(), y2_err.flatten()

def plot_gpr(x1, y1, kernel, **kwargs):
    x2, y2, y2_err = generate_gpr(x1, y1, kernel, **kwargs)
    _, ax = plt.subplots(figsize=(12,7))
    ax.scatter(x1, y1, color='purple', marker='x', alpha=0.5)
    ax.plot(x2, y2, color='red', ls='-', lw=2)
    ax.fill_between(x2, y2-y2_err, y2+y2_err, color='purple', alpha=0.3)
    ax.grid(True)
